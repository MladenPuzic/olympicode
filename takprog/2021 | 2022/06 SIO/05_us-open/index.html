#define MAXN (1<<20)+7
using namespace std;
int a[MAXN],b[MAXN],dp[MAXN],aux[MAXN];
void rek(int l,int r)
{
    if(l==r) {dp[l]=0; return;}
    int s=(l+r)/2;
    rek(l,s); rek(s+1,r);
    int t1=l,t2=r;
    int mx=0,pmax=-500;
    for(int i=s+1;i<=r;i++) mx=max(mx,dp[i]);
    for(int i=s;i>=l;i--)
    {
        while(t2!=s && a[t2]>=a[i]) pmax=max(pmax,dp[t2--]);

    }

    {
